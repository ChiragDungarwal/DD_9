<script>
  // Product URL mapping
  const productUrls = {
    'dol-starter': 'dol-starter.html',
    'fasd-starter': 'fasd-starter.html',
    'motor-starter-panel': 'motor-starter-panel.html',
    'single-phase-pump-panel': 'vfd-panel.html',
    'mccb': 'mccb.html',
    'mcb': 'mcb.html',
    'distribution-box': 'distribution-box.html',
    'sdf': 'sdf.html',
    'acb': 'acb.html',
    'rccb': 'rccb.html',
    'mini-contactor': 'mini-contactor.html',
    'contactor': 'contactor.html',
    'special-contactor': 'special-contactor.html',
    'control-relay': 'control-relay.html',
    'dc-contactor': 'dc-contactor.html',
    'overload-relay': 'overload-relay.html',
    'mpcb': 'mpcb.html',
    'electronic-overload': 'electronic-overload.html',
    'apfc-relay': 'apfc-relay.html',
    'power-capacitor': 'power-capacitor.html',
    'control-devices': 'control-devices.html',
    'foot-switch': 'foot-switch.html',
    'ac-brake': 'ac-brake.html',
    'dc-brake': 'dc-brake.html',
    'timer': 'timer.html',
    'wire': 'wire.html',
    'socket': 'socket.html',
    'limit-switch': 'limit-switch.html',
    'proximity-sensor': 'proximity-sensor.html',
    'enclosures': 'enclosures.html',
    'ats': 'ats.html',
    'changeover': 'changeover.html',
    'switch-disconnector': 'switch-disconnector.html',
    'fuse-switch': 'fuse-switch.html',
    'energy-meter': 'energy-meter.html',
    'power-analyzer': 'power-analyzer.html',
    'monitoring-system': 'monitoring-system.html',
    'power-controller': 'power-controller.html'
  };

  // Enable scroll restoration
  if ('scrollRestoration' in history) {
    history.scrollRestoration = 'manual';
  }

  // Save scroll position before leaving
  window.addEventListener('beforeunload', () => {
    sessionStorage.setItem('products_scroll_position', window.scrollY);
    
    // Save expanded categories state
    const expandedCategories = [];
    document.querySelectorAll('.category-content.active').forEach((content, index) => {
      const categoryBtn = content.previousElementSibling;
      if (categoryBtn && categoryBtn.classList.contains('category-toggle')) {
        const categoryText = categoryBtn.querySelector('span').innerText;
        expandedCategories.push(categoryText);
      }
    });
    sessionStorage.setItem('products_expanded_categories', JSON.stringify(expandedCategories));
    
    // Save expanded subcategories state
    const expandedSubcategories = [];
    document.querySelectorAll('.subcategory-content.active').forEach((content) => {
      const subcategoryBtn = content.previousElementSibling;
      if (subcategoryBtn && subcategoryBtn.classList.contains('subcategory-toggle')) {
        const subcategoryText = subcategoryBtn.querySelector('span').innerText;
        expandedSubcategories.push(subcategoryText);
      }
    });
    sessionStorage.setItem('products_expanded_subcategories', JSON.stringify(expandedSubcategories));
  });

  // Restore scroll position and expanded states on load
  window.addEventListener('load', () => {
    // Restore scroll position
    const savedScroll = sessionStorage.getItem('products_scroll_position');
    if (savedScroll) {
      setTimeout(() => {
        window.scrollTo(0, parseInt(savedScroll));
      }, 100);
    }
    
    // Restore expanded categories
    const savedCategories = sessionStorage.getItem('products_expanded_categories');
    if (savedCategories) {
      const categoriesToExpand = JSON.parse(savedCategories);
      document.querySelectorAll('.category-toggle').forEach(toggle => {
        const span = toggle.querySelector('span');
        if (span && categoriesToExpand.includes(span.innerText)) {
          toggle.click();
        }
      });
    }
    
    // Restore expanded subcategories
    const savedSubcategories = sessionStorage.getItem('products_expanded_subcategories');
    if (savedSubcategories) {
      const subcategoriesToExpand = JSON.parse(savedSubcategories);
      document.querySelectorAll('.subcategory-toggle').forEach(toggle => {
        const span = toggle.querySelector('span');
        if (span && subcategoriesToExpand.includes(span.innerText)) {
          toggle.click();
        }
      });
    }
  });

  // Category toggle functionality
  document.querySelectorAll('.category-toggle').forEach(button => {
    button.addEventListener('click', () => {
      const content = button.nextElementSibling;
      button.classList.toggle('active');
      content.classList.toggle('active');
    });
  });

  // Subcategory toggle functionality
  document.querySelectorAll('.subcategory-toggle').forEach(button => {
    button.addEventListener('click', (e) => {
      e.stopPropagation();
      const content = button.nextElementSibling;
      button.classList.toggle('active');
      content.classList.toggle('active');
    });
  });

  // Make product cards clickable - saves scroll position before navigating
  document.querySelectorAll('.product-card').forEach(card => {
    card.addEventListener('click', (e) => {
      e.preventDefault();
      
      const productKey = card.getAttribute('data-product');
      const productUrl = productUrls[productKey];
      
      if (productUrl) {
        // Save current scroll position before navigating
        sessionStorage.setItem('products_scroll_position', window.scrollY);
        
        // Save expanded categories state
        const expandedCategories = [];
        document.querySelectorAll('.category-content.active').forEach((content) => {
          const categoryBtn = content.previousElementSibling;
          if (categoryBtn && categoryBtn.classList.contains('category-toggle')) {
            const categoryText = categoryBtn.querySelector('span').innerText;
            expandedCategories.push(categoryText);
          }
        });
        sessionStorage.setItem('products_expanded_categories', JSON.stringify(expandedCategories));
        
        // Save expanded subcategories state
        const expandedSubcategories = [];
        document.querySelectorAll('.subcategory-content.active').forEach((content) => {
          const subcategoryBtn = content.previousElementSibling;
          if (subcategoryBtn && subcategoryBtn.classList.contains('subcategory-toggle')) {
            const subcategoryText = subcategoryBtn.querySelector('span').innerText;
            expandedSubcategories.push(subcategoryText);
          }
        });
        sessionStorage.setItem('products_expanded_subcategories', JSON.stringify(expandedSubcategories));
        
        // Navigate to product page
        window.location.href = productUrl;
      } else {
        console.warn('No URL mapping found for product:', productKey);
      }
    });
  });
</script>
